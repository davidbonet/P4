35 #define CTTE_GAUSSIAN -0.9189385332046727 /* log(1 / sqrt(2 * PI)) */
36 
37  /*
38  Use this function to compute log(x+y) from logx and logy:
39 
40  The basic expression,
41  log(e^logx + e^logy),
42  may result in 'nan', if e^logx or e^logy is too large
43 
44  Solution:
45  => if logx is larger,
46  log(e^logx + e^logy)=log(e^logx * (1+e^logy/e^logx)) = logx +log(1+e^(logy-logx))
47 
48  (same idea if logy is larger)
49  */
50 
51  float add_logs(float logx, float logy) {
52  if (logx > logy)
53  return logx + log(1.0 + exp(logy-logx));
54  else
55  return logy + log(1.0 + exp(logx-logy));
56  }

58  float gaussian_logprob(unsigned int vector_size, const float *mu, const float *inv_sigma, const float *x) {
59  float e=0.0F, c=0.0F, logp;
60  unsigned int j;
61  for (j=0; j<vector_size; ++j) {
62  float f = (x[j]-mu[j]) * inv_sigma[j];
63  e += (f*f);
64  c += log(inv_sigma[j]);
65  }
66  e /= 2;
67  logp = vector_size * CTTE_GAUSSIAN + c - e;
68  return logp;
69  }

91  float GMM::gmm_logprob(const float *x) const {
92  float log_prob_x,f;
93  unsigned int k;
94  log_prob_x = log(w[0]) + gaussian_logprob(vector_size, mu[0], inv_sigma[0], x);
95

140  int GMM::em_maximization(const upc::fmatrix &data, const upc::fmatrix &weights) {
141  unsigned int n, j, k;
142 
143  w.reset();
144  mu.reset();
145  inv_sigma.reset();
146 
147  for (n=0; n<data.nrow(); ++n) {
148  for (k=0; k < nmix; ++k) {
149  w[k] += weights[n][k];
150  for (j=0; j < vector_size; ++j) {
151  mu[k][j] += weights[n][k] * data[n][j]; /* sum{x w_i} */
152  inv_sigma[k][j] += weights[n][k] * data[n][j] * data[n][j]; /* sum{x^2 w_i} */
153  }
154  }
155  }
156  for (k=0; k < nmix; ++k) {
157  for (j=0; j < vector_size; ++j) {
158  mu[k][j] /= w[k]; /* sum{x w_i}/sum{w_i} */
159  inv_sigma[k][j] /= w[k]; /* sum{x^2 w_i}/sum{w_i} */
160  inv_sigma[k][j] = 1.0F/sqrt(inv_sigma[k][j] - mu[k][j]*mu[k][j]); /* 1/sigma */
161  }
162  w[k] /= data.nrow();
163  }
164  return 0;
165  }

170  float GMM::em_expectation(const fmatrix &data, fmatrix &weights) const {
171  unsigned int n, k;
172  float log_prob_total, log_prob_x;
173 
174  if (data.ncol() != vector_size)
175  return -1.0;
176 
177  if (weights.nrow() != data.nrow() or
178  weights.ncol() != nmix)
179  weights.resize(data.nrow(), nmix);
180 
181  //use log(prob) for intermediate computation, to avoid underflow
182 
183  //For each input data ...
184  for (n=0, log_prob_total = 0.0F; n<data.nrow(); ++n) {
185  //For each mixture ...
186  for (k=0, log_prob_x = -1e20F; k < nmix; ++k) {
187  weights[n][k] = log(w[k]) + gaussian_logprob(vector_size, mu[k], inv_sigma[k], data[n]);
188  log_prob_x = add_logs(log_prob_x, weights[n][k]);
189  }
190 
191  for (k=0; k < nmix; ++k)
192  weights[n][k] = exp(weights[n][k]-log_prob_x);
193  log_prob_total += log_prob_x;
194  }
195 
196  log_prob_total /= data.nrow();
197  return log_prob_total;
198  }

200  int GMM::em(const fmatrix &data, unsigned int max_it, float inc_threshold, int verbose) {
201  unsigned int iteration;
202  float old_prob=-1e34, new_prob=-1e34, inc_prob=-1e34;
203 
204  fmatrix weights(data.nrow(), nmix);
205  for (iteration=0; iteration<max_it; ++iteration) {
206  /// \HECHO
207  // Complete the loop in order to perform EM, and implement the stopping criterion.
208  //
209  // EM loop: em_expectation + em_maximization.
210  //
211  // Update old_prob, new_prob and inc_prob in order to stop the loop if logprob does not
212  // increase more than inc_threshold.
213 
214  new_prob = em_expectation(data, weights);
215  em_maximization(data,weights);
216  inc_prob = new_prob - old_prob;
217  old_prob = new_prob;
218 
219 
220  if (verbose & 01)
221  cout << "GMM nmix=" << nmix << "\tite=" << iteration << "\tlog(prob)=" << new_prob << "\tinc=" << inc_prob << endl;
222 
223  if(inc_prob <= inc_threshold)
224  return 0;
225  }
226  return 0;
227  }

238  void GMM::split_mixture(unsigned int src, unsigned int dest) {
239  unsigned int j;
240  int sign;
241  float r;
242  for (j=0; j<vector_size; ++j) {
243  r = (float) 2.0F *rand()/(float) RAND_MAX - 1.0F; /* r: (-1,1) */
244  sign = (r > 0 ? 1 : -1);
245 
246  mu[dest][j] = mu[src][j] + sign * 0.5/inv_sigma[src][j];
247  mu[src][j] = mu[src][j] - sign * 0.5/inv_sigma[src][j];
248 
249  inv_sigma[src][j] *= sqrt(2);
250  inv_sigma[dest][j] = inv_sigma[src][j];
251  }
252  w[src] /= 2.0F;
253  w[dest] = w[src];
254  }

256  int GMM::split(unsigned int target_size) {
257  unsigned int i, j, old_size;
258 
259  if (nmix >= target_size)
260  return nmix;
261 
262  if (2*nmix <= target_size) {
263  target_size = 2*nmix;
264  old_size = nmix;
265  resize(target_size, vector_size);
266  for (i=old_size, j=0; i<nmix; ++i, ++j)
267  split_mixture(j, i);
268  } else {
269  old_size = nmix;
270  resize(target_size, vector_size);
271  /* best way: select mixtures with larger variance (now, the first ones) */
272  for (i=old_size, j=0; i<nmix; ++i, ++j)
273  split_mixture(j, i);
274  }
275  return nmix;
276  }

278  int GMM::random_init(const upc::fmatrix &data, unsigned int nmix) {
279  if (data.nrow() == 0 or data.ncol() == 0)
280  return -1;
