12 def read_gmm(fileGMM):
13  '''
14  Reads the weights, means and convariances from a GMM
15  stored using format "UPC: GMM V 2.0"
16  '''
17 
18  header = b'UPC: GMM V 2.0\x00'
19 
20  try:
21  with open(fileGMM, 'rb') as fpGmm:
22  headIn = fpGmm.read(15)
23 
24  if headIn != header:
25  print('ERROR: {fileGMM} is not a valid GMM file'.format(fileGMM=fileGMM))
26  exit(-1)
27 
28  numMix = struct.unpack('@I', fpGmm.read(4))[0]
29  weights = np.array(struct.unpack('{numMix}f'.format(numMix=numMix), fpGmm.read(numMix * 4)))
30 
31  (numMix, numCof) = struct.unpack('@II', fpGmm.read(2 * 4))
32  means = struct.unpack('@{totCof}f'.format(totCof=numMix*numCof), fpGmm.read(numMix * numCof * 4))
33  means = np.array(means).reshape(numMix, numCof)
34 
35  (numMix, numCof) = struct.unpack('@II', fpGmm.read(2 * 4))
36  invStd = struct.unpack('@{totCof}f'.format(totCof=numMix*numCof), fpGmm.read(numMix * numCof * 4))
37  covs = np.array(invStd).reshape(numMix, numCof) ** -2
38 
39  return weights, means, covs
40  except:
41  raise Exception('Error al leer el fichero {fileGMM}'.format(fileGMM=fileGMM))

59 def pdfGMM(X, weights, means, covs):
60  '''
61  Returns the probability density function (PDF) of a population X
62  given a Gaussian Mixture Model (GMM) defined by its weights,
63  means and covariances.
64  '''
65 
66  pdf = np.zeros(len(X))
67  for mix, weight in enumerate(weights):
68  try:
69  pdf += weight * gauss.pdf(X, mean=means[mix], cov=covs[mix])
70  except:
71  raise Exception('Error al calcular la mezcla {mix} del GMM'.format(mix=mix))
72 
73  return pdf

75 def limsGMM(means, covs, fStd=3):
76  '''
77  Returns the maximum and minimum values of the mean plus/minus fStd
78  times the standard deviation for a set of Gaussians defined by their
79  means and convariances.
80  '''
81 
82  numMix = len(means)
83 
84  min_ = means[0][:]
85  max_ = means[0][:]
86 
87  for mix in range(numMix):
88  min_ = np.min((min_, means[mix] - fStd * covs[mix] ** 0.5), axis=0)
89  max_ = np.max((max_, means[mix] + fStd * covs[mix] ** 0.5), axis=0)
90 
91  margin = max(max_ - min_)
92 
93  return min_, max_
